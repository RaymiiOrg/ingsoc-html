The below table has systemd versus init examples. the service used as example is *nginx*:

<table class="table-striped">
	<thead>
		<tr>
			<th>Action</th>
			<th>Debian Init</th>
			<th>Ubuntu Upstart</th>
			<th>Red Hat/CentOS Init</th>
			<th><b>Systemd</b></th>
		</tr>
	</thead>
	<tbody>
		<tr>
			<td>Start service</td>
			<td><code>/etc/init.d/nginx start</code></td>
			<td><code>initctl start nginx</code><br />
			<code>/etc/init.d/nginx start</code></td>
			<td><code>/etc/init.d/nginx start</code></td>
			<td><code>systemctl start nginx.service</code></td>
		</tr>
		<tr>
			<td>Stop service</td>
			<td><code>/etc/init.d/nginx stop</code></td>
			<td><code>initctl stop nginx</code><br />
			<code>/etc/init.d/nginx stop</code></td>
			<td><code>/etc/init.d/nginx stop</code></td>
			<td><code>systemctl stop nginx.service</code></td>
		</tr>
		<tr>
			<td>Restart service</td>
			<td><code>/etc/init.d/nginx restart</code></td>
			<td><code>initctl restart nginx</code><br />
				<code>/etc/init.d/nginx restart</code></td>
			<td><code>/etc/init.d/nginx restart</code></td>
			<td><code>systemctl restart nginx.service</code></td>
		</tr>
		<tr>
			<td>Enable service at boot / startup</td>
			<td><code>update-rc.d nginx defaults</code></td>
			<td>edit <code>/etc/init/nginx.conf</code>, add <code>start on local-filesystems</code> line<br />
				<code>update-rc.d nginx defaults</code> 
			</td>
			<td><code>chkconfig nginx on</code><br />
				<code>ntsysv</code></td>
			<td><code>systemctl enable nginx.service</code></td>
		</tr>
		<tr>
			<td>Disable service at startup / boot</td>
			<td><code>update-rc.d nginx remove</code></td>
			<td><code>echo 'manual' > /etc/init/nginx.override</code><br />
			<code>update-rc.d nginx remove</code></td>
			<td><code>chkconfig nginx off</code><br />
				<code>ntsysv</code></td>
			<td><code>systemctl disable nginx.service</code></td>
		</tr>
		<tr>
			<td>List all available services</td>
			<td><code>ls /etc/init.d/</code></td>
			<td><code>initctl list</code></td>
			<td><code>chkconfig --list</code><br />
				<code>ntsysv</code></td>
			<td><code>systemctl list-units -t service --all</code></td>
		</tr>
		<tr>
			<td>Get a service status</td>
			<td><code>/etc/init.d/nginx status</code></td>
			<td><code>initctl status nginx</code></td>
			<td><code>/etc/init.d/nginx status</code></td>
			<td><code>systemctl status nginx.service</code></td>
		</tr>
